Calculator Example

import theia.math.Arithmetic;

public class Main {

    public static void main(String[] args) {

        double a = 12;
        double b = 4;

        System.out.println(
            "Add: " +
            Arithmetic.Add(a, b)
        );

        System.out.println(
            "Sub: " +
            Arithmetic.Sub(a, b)
        );

        System.out.println(
            "Mul: " +
            Arithmetic.Mul(a, b)
        );

        System.out.println(
            "Div: " +
            Arithmetic.Div(a, b)
        );

    }

}

Average of Three Numbers

import theia.math.Arithmetic;

public class Main {

    public static void main(String[] args) {

        double a = 10;
        double b = 20;
        double c = 30;

        double sum =
            Arithmetic.Add(
                Arithmetic.Add(a, b),
                c
            );

        double average =
            Arithmetic.Div(sum, 3);

        System.out.println(average);

    }

}

Error Handling

import theia.math.Arithmetic;

public class Main {

    public static void main(String[] args) {

        try {

            double result =
                Arithmetic.Div(
                    10,
                    0
                );

            System.out.println(result);

        } catch (ArithmeticException ex) {

            System.out.println(
                ex.getMessage()
            );

        }

    }

}

Mini Grade Calculator

import theia.math.Arithmetic;

public class Main {

    public static void main(String[] args) {

        double exam1 = 8;
        double exam2 = 9;
        double exam3 = 7;

        double total =
            Arithmetic.Add(
                Arithmetic.Add(
                    exam1,
                    exam2
                ),
                exam3
            );

        double average =
            Arithmetic.Div(
                total,
                3
            );

        System.out.println(
            "Average: " +
            average
        );

    }

}
